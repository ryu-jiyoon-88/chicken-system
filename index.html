packages = ["pandas", "openpyxl", "matplotlib"] import pandas as pd import matplotlib.pyplot as plt import io from js import document, Uint8Array, window, Blob from pyodide.ffi import create_proxy # matplotlib 한글 폰트 설정 (깃허브 서버 환경에 맞춤) import matplotlib.font_manager as fm font_path = fm.findfont(fm.FontProperties(family="AppleGothic" if window.navigator.platform.startswith('Mac') else "Malgun Gothic")) fm.font_set_choice = 'AppleGothic' if window.navigator.platform.startswith('Mac') else 'Malgun Gothic' plt.rcParams['font.family'] = fm.font_set_choice plt.rcParams['axes.unicode_minus'] = False # 마이너스 기호 깨짐 방지 async def process_file(event): status = document.getElementById("status") chart_area = document.getElementById("chart-area") file_list = event.target.files if len(file_list) == 0: return status.innerHTML = "⏳ 데이터를 정밀 분석하여 그래프를 그리는 중이야..." file = file_list.item(0) array_buf = await file.arrayBuffer() content = array_buf.to_py() try: # 데이터 로드 df = pd.read_excel(io.BytesIO(content)) # 수치 데이터 형변환 (매우 중요) for col in ['판매가(부가세별도)', '구매가(부가세별도)', '수량']: if col in df.columns: df[col] = pd.to_numeric(df[col], errors='coerce').fillna(0) # 정산 로직 적용 (PB 17.4% / NB 3% 예시) df['본사수익'] = df.apply(lambda x: x['판매가(부가세별도)'] * 0.174 if 'PB' in str(x.get('구분','')) else x['판매가(부가세별도)'] * 0.03, axis=1) # 가맹점명 컬럼 유무 확인 shop_col = '가맹점명' if '가맹점명' in df.columns else '매장' # --- 그래프 그리기 --- # 그래프 그리는 헬퍼 함수 def create_chart_blob(fig): out_io = io.BytesIO() fig.savefig(out_io, format='png', bbox_inches='tight') plt.close(fig) # 메모리 관리 return Blob.new([Uint8Array.new(out_io.getvalue())], { 'type': 'image/png' }) # [목차 1] 품목별 본사 수익 기여도 (상위 5개 도넛 차트) profit_by_item = df.groupby('품목명')['본사수익'].sum().nlargest(5) fig1, ax1 = plt.subplots(figsize=(7, 7)) ax1.pie(profit_by_item, labels=profit_by_item.index, autopct='%1.1f%%', startangle=140, pctdistance=0.85) centre_circle = plt.Circle((0,0),0.70,fc='white') fig1.gca().add_artist(centre_circle) ax1.axis('equal') ax1.set_title('본사 수익 기여 품목 TOP 5') chart1_url = window.URL.createObjectURL(create_chart_blob(fig1)) document.getElementById("chart1").innerHTML = f'' # [목차 2] 전체 매출액 구성 비율 (PB vs NB 파이 차트) sales_by_type = df.groupby('구분')['판매가(부가세별도)'].sum() if not sales_by_type.empty: fig2, ax2 = plt.subplots(figsize=(7, 7)) ax2.pie(sales_by_type, labels=sales_by_type.index, autopct='%1.1f%%', startangle=90) ax2.axis('equal') ax2.set_title('PB 상품 vs NB 상품 매출 비율') chart2_url = window.URL.createObjectURL(create_chart_blob(fig2)) document.getElementById("chart2").innerHTML = f'' # [목차 3] 매장별 정산 매출 순위 (TOP 5 바 차트) if shop_col in df.columns: sales_by_shop = df.groupby(shop_col)['판매가(부가세별도)'].sum().nlargest(5).sort_values() fig3, ax3 = plt.subplots(figsize=(8, 6)) bars = ax3.barh(sales_by_shop.index, sales_by_shop / 10000) # 만원 단위 ax3.set_xlabel('정산 대상 매출액 (만원)') ax3.set_title('정산 매출액 상위 5개 매장') ax3.bar_label(bars, fmt='%d', padding=5) # 값 표시 chart3_url = window.URL.createObjectURL(create_chart_blob(fig3)) document.getElementById("chart3").innerHTML = f'' # [목차 4] PB vs NB 상품 수익성 비교 (바 차트) if '구분' in df.columns: profit_by_type = df.groupby('구분')['본사수익'].sum() margin_by_type = (profit_by_type / df.groupby('구분')['판매가(부가세별도)'].sum()) * 100 fig4, ax4 = plt.subplots(figsize=(8, 6)) bars = ax4.bar(margin_by_type.index, margin_by_type) ax4.set_ylabel('평균 실마진율 (%)') ax4.set_title('상품 구분별 평균 수익률') ax4.bar_label(bars, fmt='%.1f%%', padding=5) # 값 표시 chart4_url = window.URL.createObjectURL(create_chart_blob(fig4)) document.getElementById("chart4").innerHTML = f'' # 최종 화면 표시 status.innerHTML = "✅ 분석 완료! 목차별 그래프를 확인해봐." chart_area.style.display = "grid" except Exception as e: status.innerHTML = f"❌ 오류: 엑셀 형식이 맞는지, 가맹점명이 있는지 확인해줘!

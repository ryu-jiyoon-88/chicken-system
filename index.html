packages = ["pandas", "openpyxl"] import pandas as pd import io from js import document, Uint8Array, window, Blob from pyodide.ffi import create_proxy async def process_file(event): status = document.getElementById("status") summary_area = document.getElementById("summary-area") file_list = event.target.files if len(file_list) == 0: return status.innerHTML = "⏳ 데이터를 정밀 분석 중입니다..." file = file_list.item(0) array_buf = await file.arrayBuffer() content = array_buf.to_py() try: # 1. 데이터 로드 및 전처리 df = pd.read_excel(io.BytesIO(content)) # 수치 데이터 형변환 (공백 제거 등) for col in ['판매가(부가세별도)', '구매가(부가세별도)', '수량']: if col in df.columns: df[col] = pd.to_numeric(df[col], errors='coerce').fillna(0) # 2. 정산 로직 적용 df['본사수익'] = df.apply(lambda x: x['판매가(부가세별도)'] * 0.174 if 'PB' in str(x.get('구분','')) else x['판매가(부가세별도)'] * 0.03, axis=1) df['실마진율'] = (df['본사수익'] / df['판매가(부가세별도)'].replace(0, 1)) * 100 # 3. 세 줄 요약 분석 로직 # (1) 최고 효자 제품 분석 best_item = df.groupby('품목명')['본사수익'].sum().idxmax() # (2) 최고 매출 매장 분석 (가맹점명 컬럼이 있다고 가정, 없으면 '전체'로 표시) shop_col = '가맹점명' if '가맹점명' in df.columns else '매장' if shop_col in df.columns: best_shop = df.groupby(shop_col)['판매가(부가세별도)'].sum().idxmax() shop_msg = f"🏆 가장 발주가 활발한 곳은 **'{best_shop}'** 매장입니다." else: shop_msg = "🏆 전반적으로 PB 상품군에서 높은 수익률이 관찰됩니다." # (3) 수익성 요약 total_profit = df['본사수익'].sum() avg_margin = df['실마진율'].mean() # 화면에 요약 표시 document.getElementById("line-1").innerHTML = f"📍 이번 데이터에서 본사 수익에 가장 기여한 제품은 **'{best_item}'**입니다." document.getElementById("line-2").innerHTML = shop_msg document.getElementById("line-3").innerHTML = f"💰 총 본사 예상 수익은 **{int(total_profit):,}원**이며, 평균 실마진율은 **{avg_margin:.1f}%**입니다." # 4. 엑셀 다운로드 생성 out_io = io.BytesIO() with pd.ExcelWriter(out_io, engine='openpyxl') as writer: df.to_excel(writer, index=False, sheet_name='정산분석결과') excel_data = out_io.getvalue() blob = Blob.new([Uint8Array.new(excel_data)], { 'type': 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet' }) url = window.URL.createObjectURL(blob) btn = document.getElementById("download-btn") btn.href = url btn.download = f"치킨파머스_분석결과_{file.name}" status.innerHTML = "✅ 분석 완료!" summary_area.style.display = "block" except Exception as e: status.innerHTML = f"❌ 오류: 엑셀 형식을 확인해 주세요. ({str(e)})" file_event = create_proxy(process_file) document.getElementById("file-upload").addEventListener("change", file_event)

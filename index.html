packages = ["pandas", "openpyxl", "matplotlib"] import pandas as pd import matplotlib.pyplot as plt import io from js import document, Uint8Array, window, Blob from pyodide.ffi import create_proxy # 준비 완료 시 로딩 메시지 숨기기 document.getElementById("loading-msg").style.display = "none" async def run_analysis(event): file = event.target.files.item(0) if not file: return array_buf = await file.arrayBuffer() df = pd.read_excel(io.BytesIO(array_buf.to_py())) # 숫자 데이터 정리 for c in ['판매가(부가세별도)', '구매가(부가세별도)', '수량']: if c in df.columns: df[c] = pd.to_numeric(df[c], errors='coerce').fillna(0) # 17.4%(PB) vs 3%(NB) 수익 계산 df['본사수익'] = df.apply(lambda x: x['판매가(부가세별도)'] * 0.174 if 'PB' in str(x.get('구분','')) else x['판매가(부가세별도)'] * 0.03, axis=1) def plot_to_div(fig, div_id): buf = io.BytesIO() fig.savefig(buf, format='png', bbox_inches='tight') plt.close(fig) url = window.URL.createObjectURL(Blob.new([Uint8Array.new(buf.getvalue())], { 'type': 'image/png' })) document.getElementById(div_id).innerHTML = f'' # [목차 1] 도넛 차트 top5 = df.groupby('품목명')['본사수익'].sum().nlargest(5) fig1, ax1 = plt.subplots() ax1.pie(top5, labels=top5.index, autopct='%1.1f%%', startangle=140, wedgeprops=dict(width=0.3)) plot_to_div(fig1, "graph1") # [목차 2] 파이 차트 type_sales = df.groupby('구분')['판매가(부가세별도)'].sum() fig2, ax2 = plt.subplots() ax2.pie(type_sales, labels=type_sales.index, autopct='%1.1f%%') plot_to_div(fig2, "graph2") # [목차 3] 바 차트 shop_col = '가맹점명' if '가맹점명' in df.columns else '품목명' top_shops = df.groupby(shop_col)['판매가(부가세별도)'].sum().nlargest(5) fig3, ax3 = plt.subplots() top_shops.plot(kind='barh', ax=ax3, color='#e67e22') plot_to_div(fig3, "graph3") # [목차 4] 수익성 바 차트 margin_compare = df.groupby('구분')['본사수익'].sum() fig4, ax4 = plt.subplots() margin_compare.plot(kind='bar', ax=ax4, color=['#3498db', '#2ecc71']) plot_to_div(fig4, "graph4") document.getElementById("dashboard").style.opacity = "1" file_handler = create_proxy(run_analysis) document.getElementById("file-upload").addEventListener("change", file_handler)
